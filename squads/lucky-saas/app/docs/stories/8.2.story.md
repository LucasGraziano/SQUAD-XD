## Contexto

Com o schema expandido (8.1), precisamos das server actions que toda a UI do Deal Room vai consumir. O objetivo é ter uma camada de dados limpa e única, eliminando a fragmentação entre `actions/quotes.ts`, `actions/proposals.ts` e `actions/leads.ts`.

As actions existentes em `quotes.ts` são mantidas e expandidas. `proposals.ts` é deprecado (suas responsabilidades migram para `deals.ts`).

## Technical Spec

### Interface principal

```typescript
// src/app/actions/deals.ts
'use server'

export type DealStage = 
  | 'prospecting' | 'draft' | 'sent' | 'approved'
  | 'submitted' | 'under_analysis' | 'issued' | 'contracted' | 'rejected'

export interface DealSummary {
  id: string
  stage: DealStage
  ramo: string
  source: string | null
  object_description: string | null
  notes: string | null
  share_token: string
  view_count: number
  last_viewed_at: string | null
  sent_at: string | null
  approved_at: string | null
  submitted_at: string | null
  issued_at: string | null
  contracted_at: string | null
  response_deadline: string | null
  proposal_number: string | null
  protocol_number: string | null
  policy_id: string | null
  created_at: string
  clients: { id: string; name: string; phone: string | null; email: string | null } | null
  deal_items: DealItem[]
}

export interface DealItem {
  id: string
  seguradora: string
  premium_total: number
  payment_frequency: string
  franchise_value: number | null
  coverages: string[]
  broker_note: string | null
  is_recommended: boolean
  sort_order: number
}

export interface ConvertDealData {
  numero_apolice: string
  vigencia_inicio: string
  vigencia_fim: string
  comissao_pct?: number
  premio_override?: number
}
```

### Lógica de updateDealStage

```typescript
export async function updateDealStage(
  dealId: string,
  newStage: DealStage,
  note?: string
): Promise<{ error?: string }> {
  // 1. Buscar deal atual
  // 2. Validar que newStage está em DEAL_TRANSITIONS[currentStage]
  // 3. Montar updates: { status: newStage, [timestamp_field]: now() }
  //    - 'sent'           → sent_at
  //    - 'approved'       → approved_at
  //    - 'submitted'      → submitted_at
  //    - 'issued'         → issued_at
  //    - 'contracted'     → contracted_at
  // 4. UPDATE quote_requests
  // 5. INSERT deal_stage_history { from_stage, to_stage, note }
  // 6. revalidatePath('/pipeline')
}
```
